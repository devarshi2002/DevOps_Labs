pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo "Building the application..."
                sh 'mvn clean package'   // For Java + Maven project
            }
        }

        stage('Test') {
            steps {
                echo "Running tests..."
                sh 'npm test'
            }
        }

        stage('Deploy') {
            steps {
                echo "Deploying application..."
                // Example deploy step (can be docker run, copy files, or server deploy)
                sh 'echo "Application deployed successfully!"'
            }
        }
    }
    post {
        success {
            echo "Pipeline executed successfully!"
        }
        failure {
            echo "Pipeline failed. Please check logs."
        }
    }
}
